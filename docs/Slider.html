Class: Slider

Slider(id, parentId, minValue, maxValue, initialValue, valuePrefix, valueSuffix, classes)

Example usage: this._slider = new Slider('unique_id', 'parent_id', 0, 200, 100, 'My label', '', ' cm', 'extra_css_class'); var self = this; let sliderElement = slider.getSliderElement(); // When the slider is changed sliderElement.oninput = function() { let id = self.getId(); self._slider.updateValueLabel(this.value); // Do other stuff }
